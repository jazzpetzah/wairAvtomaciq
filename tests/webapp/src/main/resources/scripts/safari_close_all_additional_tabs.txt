set defaultTimeout to 0.3tell application "System Events"	activate application "Safari"	delay defaultTimeout	tell application "Safari"		# get all tabs in safari		set tabList to tabs of window 1		# iterate in reversed order over the list of tabs		# leave the first tab opened with ("to >>>2<<< by -1")		repeat with t from (count of tabList) to 2 by -1			# get tab with index			tell item t of tabList				#close the tab				close			end tell		end repeat	end tellend tell
