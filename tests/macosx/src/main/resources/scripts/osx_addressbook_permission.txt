set defaultTimeout to 1.0set foundNotif to falsetell application "System Events"	repeat with i from 1 to 10		if process "UserNotificationCenter" exists then			set foundNotif to true			exit repeat		end if		delay defaultTimeout	end repeat	delay defaultTimeout	if foundNotif is false then		error "No permission window found"	end if	tell process "UserNotificationCenter"		click button "%s" of window 1	end tellend tell
